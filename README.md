###git setup

install git
```
	apt-get install git
	git --version
```

setup account
```
	git config --global user.name "<name>"
	git config --global user.email "<E-mail>"
```

add repository
```
	mkdir helloGit
	cd helloGit
	git init
	ls -la
```

watch status
	`git status`

clone SSH
	`git clone git@github.com:FinleyLi/xxx.git`

clone HTTPS
	`git clone https://github.com/FinleyLi/xxx.git`

select file
```
	cd /github/109.../
	git add README.md
	git commit -m 'add git use clone and push'
	git push

	password
```
